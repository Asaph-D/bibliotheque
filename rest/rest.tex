\subsection{Annexe technique REST : endpoints et métriques}
\begin{restbox}[title={REST --- Endpoints, pipeline et métriques}]
\textbf{Base URL.} Le module REST expose ses routes sur :
\begin{lstlisting}
http://localhost:8081
\end{lstlisting}

\textbf{Fichier de requêtes.} Les appels à utiliser (Thunder Client / curl) sont listés dans :
\begin{lstlisting}
rest/request.json
\end{lstlisting}

\textbf{Pipeline d'exécution.} Pour chaque appel HTTP entrant :
\begin{lstlisting}
HTTP -> Controller (@RestController) -> Service (@Service) -> Repository (Spring Data JPA) -> SQL (PostgreSQL)
\end{lstlisting}
\end{restbox}

\begin{warningbox}[title={Définition utilisée dans ce rapport}]
Dans cette annexe, \textbf{appel interne} = \textbf{appel Java} à un \textit{repository} (Spring Data JPA), car c'est ce qui reflète
le \textbf{nombre d'opérations} nécessaires pour fabriquer la réponse côté serveur.
\end{warningbox}

\subsubsection{Nombre exact d'appels internes par endpoint}
Les chiffres ci-dessous proviennent directement de l'exécution du code (interceptions AOP côté repositories).

\subsubsection{GET \texttt{/api/livres} (liste/recherche)}
\textbf{Objectif.} Récupérer des livres et éventuellement filtrer par \texttt{titre}, \texttt{genre}, \texttt{disponible}.

\textbf{Chemin code.} \texttt{LivreController.getLivres} $\rightarrow$ \texttt{LivreService.rechercher}.

\textbf{Appels internes exacts.}
\begin{itemize}[leftmargin=1.5cm]
  \item \textbf{1 appel repository} : \texttt{livreRepository.findAll()}
  \item Les filtres \texttt{titre/genre/disponible} sont ensuite appliqués \textbf{en mémoire} (streams Java), donc \textbf{0 appel interne supplémentaire}.
\end{itemize}

\textbf{Conséquence.} Même avec des paramètres de filtre, le serveur charge d'abord \textbf{tous} les livres, puis filtre côté Java.

\subsubsection{GET \texttt{/api/livres/\{id\}} (détail)}
\textbf{Chemin code.} \texttt{LivreController.getLivre} $\rightarrow$ \texttt{LivreService.findById}.

\textbf{Appels internes exacts.}
\begin{itemize}[leftmargin=1.5cm]
  \item \textbf{1 appel repository} : \texttt{livreRepository.findById(id)}
\end{itemize}

\subsubsection{POST \texttt{/api/livres/\{livreId\}/emprunter?adherentId=...}}
\textbf{Objectif.} Créer un emprunt, marquer le livre indisponible.

\textbf{Chemin code.} \texttt{LivreController.emprunterLivre} $\rightarrow$ \texttt{EmpruntService.emprunter}.

\textbf{Appels internes exacts.}
\begin{itemize}[leftmargin=1.5cm]
  \item \textbf{1} : \texttt{livreRepository.findById(livreId)}
  \item \textbf{1} : \texttt{adherentRepository.findById(adherentId)}
  \item \textbf{(appel service)} : \texttt{livreService.markAsUnavailable(livreId)} qui fait :
    \begin{itemize}[leftmargin=1.5cm]
      \item \textbf{1} : \texttt{livreRepository.findById(livreId)}
      \item \textbf{1} : \texttt{livreRepository.save(livre)}
    \end{itemize}
  \item \textbf{1} : \texttt{empruntRepository.save(emprunt)}
\end{itemize}

\textbf{Total exact} : \textbf{5 appels internes} (2 lectures + 1 lecture + 1 écriture + 1 écriture).

\subsubsection{PUT \texttt{/api/livres/emprunts/\{empruntId\}/retourner}}
\textbf{Objectif.} Clôturer un emprunt, marquer le livre disponible.

\textbf{Chemin code.} \texttt{LivreController.retournerLivre} $\rightarrow$ \texttt{EmpruntService.retourner}.

\textbf{Appels internes exacts.}
\begin{itemize}[leftmargin=1.5cm]
  \item \textbf{1} : \texttt{empruntRepository.findById(empruntId)}
  \item \textbf{(appel service)} : \texttt{livreService.markAsAvailable(livreId)} qui fait :
    \begin{itemize}[leftmargin=1.5cm]
      \item \textbf{1} : \texttt{livreRepository.findById(livreId)}
      \item \textbf{1} : \texttt{livreRepository.save(livre)}
    \end{itemize}
  \item \textbf{1} : \texttt{empruntRepository.save(emprunt)}
\end{itemize}

\textbf{Total exact} : \textbf{4 appels internes}.

\subsubsection{Nombre d'appels HTTP nécessaires pour ``composer'' une donnée (comparaison avec GraphQL)}
\paragraph{Cas ``Query 1'' (liste filtrée).}
\textbf{REST} : \textbf{1 seul appel HTTP} (\texttt{GET /api/livres?genre=...&disponible=true}).

\paragraph{Cas ``Query 2'' (livres + auteur + emprunts actifs).}
En GraphQL, tout est renvoyé en \textbf{1 requête}. En REST, avec les endpoints présents (\texttt{/api/emprunts} ajouté), la donnée est reconstructible mais nécessite plusieurs appels :
\begin{itemize}[leftmargin=1.5cm]
  \item \textbf{1 appel HTTP} pour la liste : \texttt{GET /api/livres}
  \item \textbf{N appels HTTP} pour les emprunts actifs par livre :
  \texttt{GET /api/emprunts?livreId=\{id\}\&actifs=true}
\end{itemize}
Donc, \textbf{la donnée complète de la Query 2 est reconstructible en REST} mais au prix d'un \textbf{surcoût réseau} (\(1 + N\) appels HTTP côté client) et d'une latence potentiellement plus élevée.
